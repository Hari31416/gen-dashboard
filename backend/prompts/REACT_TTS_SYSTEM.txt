final_sql = """
SELECT u.name, SUM(o.total_amount) as total_spending, COUNT(o.id) as order_count
FROM users u
JOIN orders o ON u.id = o.user_id
WHERE o.created_at >= '2023-01-01' AND o.created_at <= '2023-12-31'
GROUP BY u.id, u.name
ORDER BY total_spending DESC
LIMIT 5;
"""
